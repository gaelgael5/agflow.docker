# PITR Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking. **Référence détaillée** : `docs/superpowers/specs/2026-05-19-pitr-design.md` (spec validée, commits `7fd4e9f` + `d4fad68`) — tous les schémas SQL, signatures Python, payloads JSON et configs pgBackRest y sont fournis verbatim. Les Tasks ci-dessous référencent les sections du spec quand le bloc serait identique.

**Goal:** Remplacer le système `backup_schedules_snapshot` (`pg_dump` à intervalle court) par un vrai PITR Postgres basé sur pgBackRest, avec basebackup quotidien + archivage WAL continu + restauration à la minute via clone Postgres éphémère exposé en pgweb.

**Architecture:** Image custom `agflow-postgres:16-pitr` (postgres:16-alpine + pgBackRest) avec `archive_mode=on`. 4 services Python SRP (basebackup, restore, clone, wal_archive) qui exécutent pgBackRest via `aiodocker.exec`. Worker APScheduler dédié (`pitr_scheduler.py`) pour basebackup quotidien + cleanup TTL clone 1h + refresh WAL window 5min. 13 endpoints REST sous `/api/admin/pitr`. Frontend : section PITR refondue avec datetime picker minute en fuseau navigateur, carte clone actif, dialog config.

**Tech Stack:** Python 3.12 + asyncpg + aiodocker + apscheduler + croniter + FastAPI + structlog + pytest / Postgres 16 + pgBackRest 2.x + pgweb / Vite + React 18 + TanStack Query + shadcn/ui + i18next + Vitest

```python
# backend/src/agflow/services/pitr_config_service.py
from __future__ import annotations

from uuid import UUID

import structlog
from croniter import CroniterBadCronError, croniter

from agflow.db import pool
from agflow.schemas.pitr import PitrConfigOut

log = structlog.get_logger(__name__)


class InvalidCronError(ValueError):
    pass


async def get_config() -> PitrConfigOut:
    row = await pool.fetch_one(
        "SELECT enabled, basebackup_cron, retention_count, updated_at FROM pitr_config WHERE id = 1"
    )
    remote_ids = await pool.fetch_all(
        "SELECT remote_connection_id FROM pitr_config_remotes WHERE config_id = 1 ORDER BY remote_connection_id"
    )
    return PitrConfigOut(
        enabled=row["enabled"],
        basebackup_cron=row["basebackup_cron"],
        retention_count=row["retention_count"],
        updated_at=row["updated_at"],
        remote_connection_ids=[r["remote_connection_id"] for r in remote_ids],
    )


async def update_config(
    *,
    enabled: bool | None = None,
    basebackup_cron: str | None = None,
    retention_count: int | None = None,
    remote_connection_ids: list[UUID] | None = None,
) -> PitrConfigOut:
    if basebackup_cron is not None:
        try:
            croniter(basebackup_cron)
        except (CroniterBadCronError, ValueError) as err:
            raise InvalidCronError(f"invalid cron: {basebackup_cron}") from err

    async with pool.transaction() as conn:
        sets, params = [], []
        if enabled is not None:
            sets.append(f"enabled = ${len(params) + 1}")
            params.append(enabled)
        if basebackup_cron is not None:
            sets.append(f"basebackup_cron = ${len(params) + 1}")
            params.append(basebackup_cron)
        if retention_count is not None:
            sets.append(f"retention_count = ${len(params) + 1}")
            params.append(retention_count)
        if sets:
            await conn.execute(f"UPDATE pitr_config SET {', '.join(sets)} WHERE id = 1", *params)

        if remote_connection_ids is not None:
            await conn.execute("DELETE FROM pitr_config_remotes WHERE config_id = 1")
            for rid in remote_connection_ids:
                await conn.execute(
                    "INSERT INTO pitr_config_remotes (config_id, remote_connection_id) VALUES (1, $1)",
                    rid,
                )

    return await get_config()
```

```python
from __future__ import annotations

import aiodocker


async def docker_exec(container_name: str, cmd: list[str]) -> tuple[int, str, str]:
    """Run command in named docker container. Returns (exit_code, stdout, stderr)."""
    async with aiodocker.Docker() as docker:
        container = await docker.containers.get(container_name)
        exec_obj = await container.exec(cmd=cmd, stdout=True, stderr=True)
        async with exec_obj.start(detach=False) as stream:
            stdout_chunks: list[bytes] = []
            stderr_chunks: list[bytes] = []
            while True:
                msg = await stream.read_out()
                if msg is None:
                    break
                if msg.stream == 1:
                    stdout_chunks.append(msg.data)
                else:
                    stderr_chunks.append(msg.data)
        info = await exec_obj.inspect()
        return (
            info.get("ExitCode") or 0,
            b"".join(stdout_chunks).decode("utf-8", errors="replace"),
            b"".join(stderr_chunks).decode("utf-8", errors="replace"),
        )
```

```python
from typing import Any

from agflow.db import pool
from agflow.schemas.pitr import BasebackupPushSummary, BasebackupSummary


class BasebackupRunningError(RuntimeError):
    pass


class BasebackupNotFoundError(LookupError):
    pass


_LABEL_RE = re.compile(r"label=([0-9A-Za-z\-_]+)")


async def list_basebackups() -> list[BasebackupSummary]:
    rows = await pool.fetch_all("""
        SELECT b.*,
               coalesce(json_agg(json_build_object(
                   'remote_connection_id', p.remote_connection_id,
                   'remote_connection_name', r.name,
                   'status', p.status,
                   'pushed_at', p.pushed_at,
                   'error', p.error,
                   'size_bytes', p.size_bytes
               )) FILTER (WHERE p.id IS NOT NULL), '[]'::json) AS pushes
        FROM pitr_basebackups b
        LEFT JOIN pitr_basebackup_pushes p ON p.basebackup_id = b.id
        LEFT JOIN remote_backup_connections r ON r.id = p.remote_connection_id
        GROUP BY b.id
        ORDER BY b.started_at DESC
    """)
    return [_row_to_basebackup_summary(r) for r in rows]


async def get_basebackup(basebackup_id: UUID) -> BasebackupSummary:
    rows = await list_basebackups()  # simpler: filter
    for r in rows:
        if r.id == basebackup_id:
            return r
    raise BasebackupNotFoundError(str(basebackup_id))


async def trigger_basebackup_now(actor_user_id: UUID | None) -> UUID:
    # Garde-fou : pas deux basebackups en parallèle
    running = await pool.fetch_one(
        "SELECT id FROM pitr_basebackups WHERE status = 'running' LIMIT 1"
    )
    if running:
        raise BasebackupRunningError(str(running["id"]))

    bid = (await pool.fetch_one(
        "INSERT INTO pitr_basebackups (pgbackrest_label, started_at, status) "
        "VALUES ($1, now(), 'running') RETURNING id",
        f"pending-{datetime.utcnow().isoformat()}",
    ))["id"]

    try:
        await ensure_stanza()
        code, stdout, err = await _pg_exec(["--stanza=" + STANZA, "backup", "--type=full"])
        if code != 0:
            raise RuntimeError(f"pgbackrest backup failed: {err}")
        match = _LABEL_RE.search(stdout)
        if not match:
            # Fallback : interroger pgbackrest info pour récupérer le dernier label
            label = await _last_label_from_info()
        else:
            label = match.group(1)

        size_bytes = await _label_size_from_info(label)

        await pool.execute(
            "UPDATE pitr_basebackups SET pgbackrest_label = $2, status='ok', "
            "completed_at = now(), size_bytes = $3 WHERE id = $1",
            bid, label, size_bytes,
        )
    except Exception as exc:
        await pool.execute(
            "UPDATE pitr_basebackups SET status='failed', completed_at = now(), error = $2 WHERE id = $1",
            bid, str(exc),
        )
        raise

    # Schedule pushes
    await _seed_pushes_for_basebackup(bid)
    return bid


async def _last_label_from_info() -> str:
    code, out, _ = await _pg_exec(["--stanza=" + STANZA, "info", "--output=json"])
    info = json.loads(out)
    backups = info[0].get("backup", [])
    if not backups:
        raise RuntimeError("no backup in pgbackrest info after backup command")
    return backups[-1]["label"]


async def _label_size_from_info(label: str) -> int | None:
    code, out, _ = await _pg_exec(["--stanza=" + STANZA, "info", "--output=json"])
    info = json.loads(out)
    for b in info[0].get("backup", []):
        if b.get("label") == label:
            return int(b.get("info", {}).get("size") or 0)
    return None


async def _seed_pushes_for_basebackup(basebackup_id: UUID) -> None:
    remotes = await pool.fetch_all(
        "SELECT remote_connection_id FROM pitr_config_remotes WHERE config_id = 1"
    )
    for r in remotes:
        await pool.execute(
            "INSERT INTO pitr_basebackup_pushes (basebackup_id, remote_connection_id, status) "
            "VALUES ($1, $2, 'pending') ON CONFLICT DO NOTHING",
            basebackup_id, r["remote_connection_id"],
        )


def _row_to_basebackup_summary(r: dict[str, Any]) -> BasebackupSummary:
    pushes_raw = r["pushes"]
    if isinstance(pushes_raw, str):
        pushes_raw = json.loads(pushes_raw)
    return BasebackupSummary(
        id=r["id"],
        pgbackrest_label=r["pgbackrest_label"],
        started_at=r["started_at"],
        completed_at=r["completed_at"],
        size_bytes=r["size_bytes"],
        status=r["status"],
        error=r["error"],
        recovery_window_start=r["recovery_window_start"],
        recovery_window_end=r["recovery_window_end"],
        pushes=[BasebackupPushSummary(**p) for p in pushes_raw],
    )
```

```python
# Append au module

class BasebackupIsLastError(RuntimeError):
    pass


class PushNotFoundError(LookupError):
    pass


async def delete_basebackup(basebackup_id: UUID) -> None:
    bb = await pool.fetch_one(
        "SELECT id, pgbackrest_label FROM pitr_basebackups WHERE id = $1 AND status = 'ok'",
        basebackup_id,
    )
    if not bb:
        raise BasebackupNotFoundError(str(basebackup_id))
    count = await pool.fetch_one(
        "SELECT count(*) AS n FROM pitr_basebackups WHERE status = 'ok'"
    )
    if count["n"] <= 1:
        raise BasebackupIsLastError(str(basebackup_id))
    code, _, err = await _pg_exec([
        "--stanza=" + STANZA, "expire", f"--set={bb['pgbackrest_label']}"
    ])
    if code != 0:
        raise RuntimeError(f"pgbackrest expire failed: {err}")
    await pool.execute("DELETE FROM pitr_basebackups WHERE id = $1", basebackup_id)


async def push_basebackup(basebackup_id: UUID, remote_id: UUID) -> None:
    push = await pool.fetch_one(
        "SELECT id, status FROM pitr_basebackup_pushes "
        "WHERE basebackup_id = $1 AND remote_connection_id = $2",
        basebackup_id, remote_id,
    )
    if not push:
        raise PushNotFoundError(f"{basebackup_id}/{remote_id}")
    if push["status"] == "ok":
        # Idempotent : ne re-push pas un succès
        return
    bb = await pool.fetch_one(
        "SELECT pgbackrest_label FROM pitr_basebackups WHERE id = $1", basebackup_id,
    )
    if not bb:
        raise BasebackupNotFoundError(str(basebackup_id))

    await pool.execute(
        "UPDATE pitr_basebackup_pushes SET status = 'pushing', error = NULL WHERE id = $1",
        push["id"],
    )
    try:
        provider = await _provider_for(remote_id)
        tarball_path = await _pgbackrest_local_path(bb["pgbackrest_label"])
        remote_path, size = await provider.upload_stream(tarball_path, "pitr")
        await pool.execute(
            "UPDATE pitr_basebackup_pushes SET status='ok', pushed_at=now(), "
            "remote_path=$2, size_bytes=$3 WHERE id = $1",
            push["id"], remote_path, size,
        )
    except Exception as exc:
        await pool.execute(
            "UPDATE pitr_basebackup_pushes SET status='failed', error=$2 WHERE id = $1",
            push["id"], str(exc),
        )
        raise


async def _pgbackrest_local_path(label: str) -> str:
    # pgbackrest stocke dans /var/lib/pgbackrest/backup/agflow/<label>/
    # On peut soit tar le dossier soit utiliser pgbackrest archive-get; pour V1 on tar le dossier
    return f"/var/lib/pgbackrest/backup/{STANZA}/{label}"


async def _provider_for(remote_id: UUID):
    from agflow.services.remote_backup_connections_service import fetch_credentials, get_connection
    from agflow.services.remote_backup_providers.factory import provider_for_kind
    conn = await get_connection(remote_id)
    creds = await fetch_credentials(conn)
    return provider_for_kind(conn.kind, creds)


async def _push_to_remotes(basebackup_id: UUID) -> None:
    pending = await pool.fetch_all(
        "SELECT remote_connection_id FROM pitr_basebackup_pushes "
        "WHERE basebackup_id = $1 AND status = 'pending'",
        basebackup_id,
    )
    for r in pending:
        try:
            await push_basebackup(basebackup_id, r["remote_connection_id"])
        except Exception as exc:  # On loggue mais on continue avec les autres remotes
            log.error("pitr.push_failed",
                      basebackup_id=str(basebackup_id),
                      remote_id=str(r["remote_connection_id"]),
                      error=str(exc))


async def _prune_old_basebackups(retention_count: int) -> int:
    """Supprime les basebackups les plus anciens au-delà de retention_count.
    Délègue à pgbackrest expire + sync de la table.
    """
    code, _, err = await _pg_exec([
        "--stanza=" + STANZA, "expire",
        f"--repo1-retention-full={retention_count}",
    ])
    if code != 0:
        raise RuntimeError(f"pgbackrest expire failed: {err}")

    # Sync DB : retirer les rows dont le label n'existe plus dans pgbackrest info
    code, out, _ = await _pg_exec(["--stanza=" + STANZA, "info", "--output=json"])
    info = json.loads(out)
    alive_labels = {b["label"] for b in info[0].get("backup", [])}

    result = await pool.fetch_one(
        "WITH deleted AS ("
        "  DELETE FROM pitr_basebackups WHERE status='ok' AND pgbackrest_label <> ALL($1::text[]) RETURNING id"
        ") SELECT count(*) AS n FROM deleted",
        list(alive_labels),
    )
    return int(result["n"])
```

```python
# backend/src/agflow/services/pitr_wal_archive_service.py
from __future__ import annotations

import json
from datetime import datetime, timezone

import structlog

from agflow.db import pool
from agflow.docker.exec_helper import docker_exec
from agflow.schemas.pitr import WalStatus

log = structlog.get_logger(__name__)
POSTGRES_CONTAINER = "agflow-postgres"
STANZA = "agflow"


async def _pg_exec(args: list[str]) -> tuple[int, str, str]:
    return await docker_exec(POSTGRES_CONTAINER, args)


async def _archive_mode_on() -> bool:
    row = await pool.fetch_one(
        "SELECT setting FROM pg_settings WHERE name = 'archive_mode'"
    )
    return row and row["setting"] == "on"


async def get_wal_status() -> WalStatus:
    code, info_out, _ = await _pg_exec(["pgbackrest", "--stanza=" + STANZA, "info", "--output=json"])
    archiving = await _archive_mode_on()

    last_archived_at = None
    archive_lag_seconds = None
    if code == 0 and info_out.strip():
        info = json.loads(info_out)
        archives = info[0].get("archive", []) if info else []
        if archives:
            # pgbackrest fournit min/max WAL files mais pas timestamp direct ;
            # on lit le mtime du dernier WAL du repo
            code2, ts_out, _ = await _pg_exec([
                "sh", "-c",
                "stat -c %Y /var/lib/pgbackrest/archive/agflow/*/" + archives[-1]["max"] + ".gz 2>/dev/null | tail -n 1"
            ])
            if code2 == 0 and ts_out.strip().isdigit():
                last_archived_at = datetime.fromtimestamp(int(ts_out.strip()), tz=timezone.utc)
                archive_lag_seconds = int((datetime.now(timezone.utc) - last_archived_at).total_seconds())

    # df pour disk usage
    code, df_out, _ = await _pg_exec(["df", "-B", "1", "/var/lib/pgbackrest"])
    used, free = 0, 0
    if code == 0:
        lines = df_out.strip().splitlines()
        if len(lines) >= 2:
            parts = lines[1].split()
            if len(parts) >= 4:
                used = int(parts[2])
                free = int(parts[3])

    return WalStatus(
        archiving_enabled=archiving,
        last_archived_at=last_archived_at,
        archive_lag_seconds=archive_lag_seconds,
        wal_disk_used_bytes=used,
        wal_disk_free_bytes=free,
    )


async def refresh_recovery_windows() -> int:
    code, info_out, _ = await _pg_exec(["pgbackrest", "--stanza=" + STANZA, "info", "--output=json"])
    if code != 0 or not info_out.strip():
        return 0
    info = json.loads(info_out)
    archives = info[0].get("archive", []) if info else []
    if not archives:
        return 0

    # On utilise un timestamp proxy : now() (acceptable car la fenêtre WAL est continue jusqu'à maintenant)
    now_utc = datetime.now(timezone.utc)
    result = await pool.fetch_one(
        "UPDATE pitr_basebackups SET recovery_window_end = $1 "
        "WHERE status='ok' RETURNING id",
        now_utc,
    )
    # fetch_one retourne la première ligne ; on veut compter
    count = await pool.fetch_one(
        "SELECT count(*) AS n FROM pitr_basebackups WHERE status='ok' AND recovery_window_end = $1",
        now_utc,
    )
    return int(count["n"])
```

```python
async def get_restore_window() -> RestoreWindow:
    row = await pool.fetch_one(
        "SELECT min(recovery_window_start) AS earliest, max(recovery_window_end) AS latest "
        "FROM pitr_basebackups WHERE status = 'ok' "
        "AND recovery_window_start IS NOT NULL AND recovery_window_end IS NOT NULL"
    )
    if not row or row["earliest"] is None or row["latest"] is None:
        raise RestoreWindowEmptyError()
    return RestoreWindow(earliest=row["earliest"], latest=row["latest"])


async def start_clone(target_time: datetime, actor_user_id: UUID | None) -> UUID:
    win = await get_restore_window()
    if target_time < win.earliest or target_time > win.latest:
        raise InvalidTargetTimeError(
            f"target_time {target_time} out of window [{win.earliest}, {win.latest}]"
        )

        "SELECT id FROM pitr_clones WHERE status IN ('restoring','ready','terminating') LIMIT 1"

        raise CloneAlreadyActiveError(str(active["id"]))

        "SELECT id, pgbackrest_label FROM pitr_basebackups "
        "WHERE status='ok' AND recovery_window_end >= $1 "
```

```python
    docker = _aiodocker()
    try:
        await docker.networks.create({"Name": network_name, "Driver": "bridge"})
        created["network"] = True
        await docker.volumes.create({"Name": volume_name})

            "Env": [

            },

        pg_container = await docker.containers.create_or_replace(name=clone_pg_name, config=pg_cfg)
        await pg_container.start()
        created["pg"] = True

        # Wait healthcheck (pg_isready dans la boucle)
        if not await _wait_pg_ready(pg_container, timeout_s=HEALTHCHECK_TIMEOUT_S):
            raise RuntimeError("postgres clone did not become ready within timeout")

        # Container pgweb

            "Image": PGWEB_IMAGE,
            "Env": [f"DATABASE_URL=postgres://agflow:agflow@{clone_pg_name}:5432/agflow?sslmode=disable"],
            "HostConfig": {
                "NetworkMode": network_name,
                "PortBindings": {"8081/tcp": [{"HostPort": ""}]},
                "AutoRemove": False,
            },

        pgweb_container = await docker.containers.create_or_replace(name=clone_pgweb_name, config=pgweb_cfg)
        await pgweb_container.start()

async def _wait_pg_ready(container, timeout_s: int) -> bool:
    elapsed = 0
    while elapsed < timeout_s:
        exec_obj = await container.exec(cmd=["pg_isready", "-U", "agflow"], stdout=True, stderr=True)
        async with exec_obj.start(detach=False) as stream:
            await stream.read_out()
        info = await exec_obj.inspect()
        if info.get("ExitCode") == 0:
            return True
        await asyncio.sleep(2)
        elapsed += 2
    return False

            pass
    try:
        await (await docker.volumes.get(volume_name)).delete()
    except aiodocker.exceptions.DockerError:
        pass
    try:
        await (await docker.networks.get(network_name)).delete()
    except aiodocker.exceptions.DockerError:
        pass
```

```python
async def get_active_clone() -> CloneStatus | None:
    row = await pool.fetch_one(
        "SELECT c.*, b.pgbackrest_label "
        "FROM pitr_clones c JOIN pitr_basebackups b ON b.id = c.basebackup_id "
        "WHERE c.status IN ('restoring','ready','terminating') LIMIT 1"
    )
    if not row:
        return None
    return _row_to_clone_status(row)

    active = await get_active_clone()
    if not active or active.status == "terminated":
        raise NoActiveCloneError()
    await pool.execute(
        "UPDATE pitr_clones SET expires_at = expires_at + INTERVAL '24 hours' WHERE id = $1",

    )
    return await get_active_clone()


async def terminate_active_clone() -> None:
    row = await pool.fetch_one(
        "SELECT id, postgres_container_name, pgweb_container_name "
        "FROM pitr_clones WHERE status IN ('restoring','ready','terminating') LIMIT 1"
    )
    if not row:
        raise NoActiveCloneError()
    await pool.execute(
        "UPDATE pitr_clones SET status='terminating' WHERE id = $1", row["id"]
    )

    docker = _aiodocker()
    try:
        await _stop_and_remove(docker, row["pgweb_container_name"])
        await _stop_and_remove(docker, row["postgres_container_name"])
        # Volume + network nommés selon convention "agflow-pitr-clone-<short>"
        if row["postgres_container_name"]:
            short = row["postgres_container_name"].removeprefix("agflow-pitr-clone-")
            await _remove_volume(docker, f"agflow-pitr-clone-data-{short}")
            await _remove_network(docker, f"pitr-clone-net-{short}")
        await pool.execute(
            "UPDATE pitr_clones SET status='terminated', terminated_at=now() WHERE id = $1",
            row["id"],

    finally:
        await docker.close()
```

```python
from __future__ import annotations

import structlog
from apscheduler.schedulers.asyncio import AsyncIOScheduler
from apscheduler.triggers.cron import CronTrigger
from apscheduler.triggers.interval import IntervalTrigger

from agflow.services import pitr_basebackup_service, pitr_clone_service, pitr_config_service, pitr_wal_archive_service

log = structlog.get_logger(__name__)
_scheduler: AsyncIOScheduler | None = None


async def start() -> None:
    global _scheduler
    if _scheduler is not None:
        return
    _scheduler = AsyncIOScheduler()
    config = await pitr_config_service.get_config()

    if config.enabled:
        _scheduler.add_job(
            _run_daily_basebackup,
            CronTrigger.from_crontab(config.basebackup_cron),
            id="pitr_daily_basebackup",
            max_instances=1,
            coalesce=True,
        )

    _scheduler.add_job(
        _run_clone_cleanup,
        IntervalTrigger(hours=1),
        id="pitr_clone_cleanup",
    )
    _scheduler.add_job(
        _run_wal_refresh,
        IntervalTrigger(minutes=5),
        id="pitr_wal_refresh",
    )
    _scheduler.start()
    log.info("pitr.scheduler_started", cron=config.basebackup_cron, enabled=config.enabled)


async def stop() -> None:
    global _scheduler
    if _scheduler:
        _scheduler.shutdown(wait=False)
        _scheduler = None


async def reload_basebackup_schedule() -> None:
    if _scheduler is None:
        return
    try:
        _scheduler.remove_job("pitr_daily_basebackup")
    except Exception:
        pass
    config = await pitr_config_service.get_config()
    if config.enabled:
        _scheduler.add_job(
            _run_daily_basebackup,
            CronTrigger.from_crontab(config.basebackup_cron),
            id="pitr_daily_basebackup",
            max_instances=1,
            coalesce=True,
        )
```

```typescript
import { useMutation, useQuery, useQueryClient } from "@tanstack/react-query";
import { pitrApi, PitrConfigUpdate, CloneStatusValue } from "@/lib/pitrApi";

const KEYS = {
  config: ["pitr", "config"] as const,
  basebackups: ["pitr", "basebackups"] as const,
  walStatus: ["pitr", "wal-status"] as const,
  restoreWindow: ["pitr", "restore-window"] as const,
  activeClone: ["pitr", "clones", "active"] as const,
};

export function usePitrConfig() {
  const qc = useQueryClient();
  const query = useQuery({ queryKey: KEYS.config, queryFn: pitrApi.getConfig });
  const update = useMutation({
    mutationFn: (payload: PitrConfigUpdate) => pitrApi.updateConfig(payload),
    onSuccess: () => qc.invalidateQueries({ queryKey: KEYS.config }),
  });
  return { ...query, update };
}

export function usePitrBasebackups() {
  return useQuery({ queryKey: KEYS.basebackups, queryFn: pitrApi.listBasebackups, refetchInterval: 30_000 });
}

export function usePitrWalStatus() {
  return useQuery({ queryKey: KEYS.walStatus, queryFn: pitrApi.getWalStatus, refetchInterval: 10_000 });
}

export function usePitrRestoreWindow() {
  return useQuery({
    queryKey: KEYS.restoreWindow,
    queryFn: pitrApi.getRestoreWindow,
    refetchInterval: 30_000,
    retry: false,
  });
}

function refetchIntervalForClone(status: CloneStatusValue | undefined): number | false {
  if (!status) return 5_000;
  if (status === "restoring" || status === "terminating") return 2_000;
  if (status === "ready") return 30_000;
  return false; // terminated, failed
}

export function usePitrActiveClone() {
  return useQuery({
    queryKey: KEYS.activeClone,
    queryFn: pitrApi.getActiveClone,
    refetchInterval: (q) => refetchIntervalForClone(q.state.data?.status),
  });
}
```

```tsx
  const d = new Date(iso);
  const yyyy = d.getFullYear();
  const mm = String(d.getMonth() + 1).padStart(2, "0");
  const dd = String(d.getDate()).padStart(2, "0");
  const hh = String(d.getHours()).padStart(2, "0");
  const mi = String(d.getMinutes()).padStart(2, "0");
  return { date: `${yyyy}-${mm}-${dd}`, time: `${hh}:${mi}` };
```
